program 		        = {function_def} statement_list

function_def 	        = DEF ID LPAREN param_list RPAREN ARROW type COLON statement_list
param_list 	            = EPSILON | param {COMMA param}
param 		            = ID COLON type
type 			        = INT_TYPE | BOOL_TYPE | LIST_TYPE LBRACK INT_TYPE RBRACK

statement_list 	        = [statement_list] statement
statement 		        = assignment_statement | if_statement | while_statement | return_statement
assignment_statement 	= ID EQ expression
if_statement 		    = IF expression COLON statement_list [ELSE COLON statement_list]
while_statement 		= WHILE expression COLON statement_list
return_statement		= RETURN expression

expression 		        = boolean_expression | or_expression

boolean_expression      = boolean_term {OR boolean_term}
boolean_term            = boolean_factor {AND boolean_factor}
boolean_factor          = NOT boolean_factor | TRUE | FALSE

or_expression           = and_expression {OR and_expression}
and_expression          = comparison {AND comparison}
comparison              = arith_expression {(LESSEQ | LESS | GREATER | GREATEREQ | EQOP | NEQ) arith_expression}
arith_expression        = term {(PLUS | MINUS) term}
term                    = factor {(TIMES | DIVIDE | MODULO) factor}
factor 		            = MINUS factor | INT | ID | ID LPAREN args_list RPAREN | LPAREN expression RPAREN | list_literal | len_expression

len_expression          = LEN LPAREN expression RPAREN

list_literal	        = LBRACK [expression {COMMA expression}] RBRACK
args_list 	            = EPSILON | expression {COMMA expression}

EPSILON 	            = <Nothing>
